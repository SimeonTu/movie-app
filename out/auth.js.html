/**
 * Secret key used for JWT token generation.
 * @constant {string}
 */
const jwtSecret = "your_jwt_secret";

const jwt = require("jsonwebtoken");
const passport = require("passport");

// Require local passport file for authentication strategy
require("./passport");

/**
 * Generates a JWT token for the given user.
 * @param {Object} user - User object for which the token is generated.
 * @returns {string} - JWT token.
 */
let generateJWTToken = (user) => {
  return jwt.sign(user, jwtSecret, {
    subject: user.Username,
    expiresIn: "7d",
    algorithm: "HS256",
  });
};

const cors = require("cors");
let allowedOrigins = ["https://ifdbase.netlify.app", "http://localhost:1234", "http://localhost:4200", "https://simeontu.github.io"];

passport.use(
  cors({
    origin: (origin, callback) => {
      if (!origin) return callback(null, true);
      if (allowedOrigins.indexOf(origin) === -1) {
        let message =
          "The CORS policy for this application doesn’t allow access from origin " +
          origin;
        return callback(new Error(message), false);
      }
      return callback(null, true);
    },
  })
);

/**
 * Express route to handle user login.
 * @module login
 * @param {Object} router - Express router object.
 */
module.exports = (router) => {
  /**
   * POST login route.
   * @name POST /login
   * @function
   * @memberof module:login
   * @param {Object} req - Express request object.
   * @param {Object} res - Express response object.
   */
  router.post("/login", (req, res) => {
    passport.authenticate("local", { session: false }, (err, user, info) => {
      if (err || !user) {
        return res.status(400).json({
          message: info.message,
          user: user,
        });
      }
      req.login(user, { session: false }, (err) => {
        if (err) {
          res.json({ error: err });
        }
        let token = generateJWTToken(user.toJSON());

        user = user.toObject(); // removes the password from the returned user object
        delete user.Password;

        return res.json({ user, token });
      });
    })(req, res);
  });
};
